Full initialization config example

Savory.js default config values can be overwritten by setting savoryConfig property at window object.

You need to set it before Savory.js script tag will be included.

<script type="text/javascript">
    window.savoryConfig = {
        // selector of main container
        container : 'container',
        // id of scripts container
        scriptsContainer : 'scripts',
        // automatically init Savory after load
        autoInit : true,
        // scroll to page top after new page loaded (emulate default browser behaviour)
        updateScrollPosition : true, 
        links : {
            // 'all links are external' mode enabled
            external : false,
            // classname to identificate ignored link nodes
            identificatorClassName : 'external'
        },
        error : {
            // className for errors container dom node for custom styling
            containerClassname : 'savory-error'
        },
        request : {
            // prevent request caching by passing timestamp to requests
            noCache : true,
            // enable passing of additional parameters to request url
            requiredParam : true,
            // additional parameters and its values list
            params : {
                savory : true
            }
        },
        callbacks : {
            // callback executed when Savory initialized (first load)
            onReady : function(savory){
            },
            // callback executed when new page is loaded
            onLoad : function(){
            }
        }
    };
</script>

Example of simple config and correct Savory.js script tag insertion:

<script type="text/javascript">
    window.savoryConfig = {
        callbacks : {
            // callback executed when Savory initialized (first load)
            onReady : function(savory){
                console.log('Savory is ready');
                // initialize your scripts
            },
            // callback executed when new page is loaded
            onLoad : function(){
                console.log('new page loaded');
                // execute your scripts
            }
        }
    };
</script>
